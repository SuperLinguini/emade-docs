[docs]    def create_representation(self, adfs=0):
        """Creates a GP tree object with primitives and automatically defined functions (ADFs)

        Args:
            adfs: Numbers of automatically defined functions (ADFs) or number of gp.PrimitiveSetTyped object primitives created.
        """
        self.psets = []
        self.exprs = []

        # Creates adfs number of gp.PrimitiveSetTyped objects. These are GP trees that serves as functions within
        # the main GP tree
        for i in np.arange(adfs):
            # Create a name for the adf, or automatically defined functions
            adf_name = 'adf_' + str(i)
            expr_name = 'expr_' + adf_name
            # Create the adf using strongly typed genetic programming
            this_adf = gp.PrimitiveSetTyped(adf_name, [GTMOEPDataPair], GTMOEPDataPair)
            # Add the primitives and terminals to this adf
            gpFrameworkHelper.addPrimitives(this_adf)
            gpFrameworkHelper.addTerminals(this_adf)
            # Add the previous adfs to this adf
            for adf in self.psets:
                this_adf.addADF(adf)
            # Add the adf to the list of adfs
            self.psets.append(this_adf)
            self.exprs.append(expr_name)
            # We need an expression object in the toolbox for each adf. Registers a GP tree creator.
            self.toolbox.register(expr_name, gp.genFull,
                 pset=this_adf, type_=this_adf.ret, min_=1, max_=3)

            # We also need a toolbox function for the creation of each adf. Uses tools.initIterate to fill the
            # PrimitiveTree with the results of the generator from expr_name
            self.toolbox.register(adf_name, tools.initIterate, gp.PrimitiveTree, getattr(self.toolbox, expr_name))

        # Initializes MAIN overarching primitive
        self.pset = gp.PrimitiveSetTyped('MAIN', [GTMOEPDataPair], GTMOEPDataPair)
        # Add the primitives and terminals to the pset
        # Primitives must be added BEFORE terminals for STGP (strongly typed genetic programming)
        gpFrameworkHelper.addPrimitives(self.pset)
        gpFrameworkHelper.addTerminals(self.pset)

        # Add the previous adfs to the final pset
        for adf in self.psets:
            self.pset.addADF(adf)
        # Reverse the psets and adfs from highest level to lowest so they can be compiled?
        self.psets = [self.pset] + self.psets[::-1]


        # We need an expression object in the toolbox for the main tree. Min height is 1 and max height is 5
        self.toolbox.register('expr_main', gp.genHalfAndHalf,
                 pset=self.pset, type_=self.pset.ret, min_=1, max_=5)

        self.exprs = ['expr_main'] + self.exprs[::-1]

        # We also need a toolbox function for the creation of the main tree
        self.toolbox.register('MAIN', tools.initIterate, gp.PrimitiveTree, self.toolbox.expr_main)

        self.toolbox.register("compile", gp.compileADF, psets=self.psets)

        # This cycle will be used to create the individual
        self.create_cycle = [self.toolbox.MAIN] + [getattr(self.toolbox, 'adf_' + str(i)) for i in np.arange(adfs-1,-1,-1)]



[docs]    def setObjectives(self, objectivesDict):
        """Builds fitness objects based on objectives dictionary

        Args:
            objectivesDict: Objectives dictionary containing information about weights and thresholds
        """
        self.objectivesDict = objectivesDict

        # Compute arrays of weights and thresholds from objective information
        weights = tuple([objectivesDict[objective]['weight'] for objective in objectivesDict])
        goals = tuple([objectivesDict[objective]['goal'] for objective in objectivesDict])
        achievable = tuple([objectivesDict[objective]['achievable'] for objective in objectivesDict])

        # Define LROI to be hypercube defined by origin and the following point,
        # minimum acceptable objective values
        self.LROI = np.array(goals)

        # Define SROI to be hypercube defined by origin and the following point,
        # region describing where solutions are known to be possible
        self.SROI = np.array(achievable)

        # Uses creator to create a FitnessMin class using the weights for base.Fitness
        creator.create("FitnessMin", base.Fitness, weights=weights)

[docs]    def buildClassifier(self, machineLearnerPath):
        """Sets up the GP framework to build a classifier

        Args:
            machineLearnerPath: path to machine learner

        """
        for dataset in self.datasetDict:
            if self.datasetDict[dataset]['type'] == 'featuredata':
                load_function = load_feature_data_from_file
            elif self.datasetDict[dataset]['type'] == 'streamdata':
                load_function = load_many_to_one_from_file
            elif self.datasetDict[dataset]['type'] == 'filterdata':
                load_function = load_many_to_many_from_file
            else:
                raise NotImplementedError(self.datasetDict[dataset]['type'] +
                        ' must be featuredata, streamdata, or filterdata')
            train_data_array = [load_function(train_data)
                                for train_data in
                            self.datasetDict[dataset]['trainFilenames']
                            ]
            test_data_array = [load_function(test_data)
                            for test_data in
                            self.datasetDict[dataset]['testFilenames']
                            ]
            # Copy the truth data in to its own location
            truth_data_array = [np.array([instance.get_target()[0]
                                for instance in test_data.get_instances()])
                                for test_data in test_data_array]

            # Clear out the truth data from the test data
            [[instance.set_target(np.array([float('NaN')]))
                                for instance in test_data.get_instances()]
                                for test_data in test_data_array]

            self.datasetDict[dataset]['dataPairArray'] = [GTMOEPDataPair(
                                                        train_data, test_data
                                                        ) for train_data, test_data in
                                                        zip(train_data_array, test_data_array)]
            self.datasetDict[dataset]['truthDataArray'] = truth_data_array

        methods.setPath(machineLearnerPath)

[docs]def my_str(individual):
    """Return the string representation of an individual

    Args:
        individual: The individual in question

    Returns:
        the string representation of an individual
    """
    # First we get the string of the main tree
    my_string = str(individual[0])

    # Next check to see if that tree uses any ADF's
    adf_matches = re.findall(r"adf_(\d+)", my_string)

    # Get the int representing the index from the back of the array e.g. [main, 2, 1, 0]
    adf_matches = np.unique([int(adf_match) for adf_match in adf_matches])
    while len(adf_matches) > 0:
        for adf_match_num in adf_matches:
            my_string += '\nadf_' + str(adf_match_num) + ': ' + str(individual[-1-adf_match_num])
        adf_matches = re.findall(r"adf_(\d+)", str(individual[-1-adf_match_num]))
        adf_matches = np.unique([int(adf_match) for adf_match in adf_matches if 'adf_' + adf_match + ': ' not in my_string])


    return my_string

[docs]def almost_equal(val1, val2, accuracy=0.01):
    """Compares val1 to val2 and returns true if within accuracy
    % (default 0.01%)

    Args:
        val1: First value
        val2: Second value

    Returns:
        boolean whether vals are within accuracy specified
    """
    return abs(val1 - val2)/val1 < accuracy/100.0



[docs]def my_similar_phenotype(ind1, ind2):
    """Defines whether two individuals are similar or not
    based upon objective scores only if strings do not match.

    Args:
        ind1: First individual
        ind2: Second individual

    Returns:
        boolean whether individuals are similar
    """
    if my_str(ind1) == my_str(ind2):
        return True
    elif ind1.fitness.values == ind2.fitness.values:
        return True
    # Check to see if all values accurate to 0.01%
    elif all(almost_equal(*values) for values in zip(ind1.fitness.values, ind2.fitness.values)):
        return True
    else:
        return False



[docs]def mate(offspring, mating_function, CXPB):
    """Iterates through a list of offspring and applies the given mating type
    With the given crossover probability

    Args:
        offspring: List of individuals
        mating_function: Function to mate
        CXPB: Probability to mate

    Returns:
        Number of individuals modified
    """
    count = 0
    for child1, child2 in zip(offspring[::2], offspring[1::2]):
        for tree1, tree2 in zip(child1, child2):
            if random.random() < CXPB:
                mating_function(tree1, tree2)

                del child1.fitness.values
                del child2.fitness.values
                for dataset in _inst.datasetDict:
                    del getattr(child1,
                        _inst.datasetDict[dataset]['name']).values
                    del getattr(child2,
                        _inst.datasetDict[dataset]['name']).values

                for objective in _inst.objectivesDict:
                    setattr(child1, _inst.objectivesDict[objective]['name'], np.ndarray(0))
                    setattr(child2, _inst.objectivesDict[objective]['name'], np.ndarray(0))

                child1.age = 0
                child2.age = 0
                count += 2
    return count




[docs]def mutate(offspring, mutate_function, MUTPB, needs_pset=False, needs_expr=False):
    """Iterates through a list of offspring and applies the given mutation
    With the given probability

    Args:
        offspring: List of individuals
        mating_function: Function to mate
        MUTPB: mutation probability

    Returns:
        Number of individuals modified
    """
    count = 0
    for mutant in offspring:
        for tree, pset, expr in zip(mutant, _inst.psets, _inst.exprs):
            if random.random() < MUTPB:
                if needs_pset and needs_expr:
                    mutate_function(tree, getattr(_inst.toolbox, expr), pset)
                elif needs_pset:
                    mutate_function(tree, pset)
                else:
                    mutate_function(tree)

                del mutant.fitness.values

                for dataset in _inst.datasetDict:
                    del getattr(mutant,
                        _inst.datasetDict[dataset]['name']).values

                for objective in _inst.objectivesDict:
                    setattr(mutant, _inst.objectivesDict[objective]['name'], np.ndarray(0))

                mutant.age = 0
                count += 1
    return count

[docs]def HCDScaleMin(individual):
    """Function to scale the fitness scores of an individual by its distance to
    the hypercube defining the region of interest

    Args:
        individual: the individual in question

    Returns:
        crowding distance
    """
    # Define LROI to be hypercube defined by origin and the following point,
    # minimum acceptable objective values
    LROI = _inst.LROI

    # Define SROI to be hypercube defined by origin and the following point,
    # region describing where solutions are known to be possible
    SROI = _inst.SROI

    myFitness = np.array(individual.fitness.values)
    q = len(myFitness)
    alpha = 0.5
    # Holder coefficient p
    p = 2.0
    HCD = 0.0
    for j in np.arange(q):
        # If fitness is worse than known solution fitness
        if myFitness[j] >= SROI[j]:
            HCD += 1 ** p

        # Fitness is somewhere between known solutions and acceptable
        # objectives
        elif myFitness[j] > LROI[j]:
            HCD += ((myFitness[j] - LROI[j]) / (SROI[j] - LROI[j])) ** p

        # Fitness is better than acceptable objectives
        elif myFitness[j] <= LROI[j]:
            HCD += 0 ** p

        else:
            print('Something is not quite right')
            sys.stdout.flush()
    HCD = HCD ** (1.0 / p)
    HCD = (alpha + HCD) / (1.0 + alpha)
    GHCD = 1.0 / HCD
    # We are trying to maximize crowding distance
    newCrowdingDist = individual.fitness.crowding_dist * GHCD

    return newCrowdingDist

    """Returns a string representation of a queue of strings

    Args:
        my_queue: queue to generate stack from

    Returns:
        stack from queue
    """
    stack_str =  ['Stack Trace:']

[docs]def handleWorker(func, dataPair, return_dict, my_queue):
    """
    Method to wrap a function evaluation.
    Required for memory monitoring and
    management.
    Takes in a function to evaluate (compiled indiviudal), a
    data pair object, and a shared return dictionary where the result

            error_string += '----------' + str(traceback.format_exc()) + '\n'
            monte_carlo_objectives = fatal_fitness
            # fatal fitness kills scores anyway
    # Average monte carlo results across number of trials
    monte_carlo_objectives = monte_carlo_objectives/len(_inst.datasetDict[dataset]['truthDataArray'])
    # Check for any infs, make fatal if any
    if np.inf in monte_carlo_objectives:
        monte_carlo_objectives = fatal_fitness
    else:
        # Let's only increment age if not fatal
        individual.age = np.floor(individual.age + 1)
    individual.fitness.values = tuple(monte_carlo_objectives)
    setattr(individual, _inst.datasetDict[dataset]['name'],
        cp.deepcopy(individual.fitness))


    # Reassign stdout and stderr
    sys.stdout = sys_stdout
    sys.stderr = sys_stderr
    return (individual, time.time() - start_time, error_string)

            individual.fitness.values = list(map(sum, zip(

                (offset, ) * len(getattr(individual,
                    _inst.datasetDict[dataset]['name']).values)))
        offset -= 1*multiplier

    return individual
