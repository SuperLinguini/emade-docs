[docs]class TriState(int):
    """Class which can take on three possible values.
    """
    def __new__(cls, val):
        if val not in (0, 1, 2):
            raise ValueError('Value must be 0, 1, or 2')
        else:
            return super(TriState, cls).__new__(cls, val)

[docs]def learnerGen():
    """Generates a machine learner object

    Returns:
        a learner object
    """
    # learners = ['KNN', 'SVM', 'Bayes', 'Trees', 'RandForest', 'Boosting', 'LogR']
    learners = ['KNN', 'SVM', 'Bayes', 'DNN', 'Trees', 'RandForest', 'Boosting', 'LogR']
    selectedLearner = random.choice(learners)
    learnerParams = None
    if selectedLearner == 'KNN':
        learnerParams = {'K': 3, 'weights':0}
    elif selectedLearner == 'SVM':
        learnerParams = {'C':1.0, 'kernel':0}
    elif selectedLearner == 'DNN':
        learnerParams = {'epochs':100, 'batch_size':128}
    elif selectedLearner == 'RandForest':
        learnerParams = {'n_estimators': 100, 'class_weight':0, 'criterion':0}
    elif selectedLearner == 'Trees':
        learnerParams = {'criterion':0, 'splitter':0}
    elif selectedLearner == 'Boosting':
        learnerParams = {'learning_rate': 0.1, 'n_estimators': 100, 'max_depth': 3}
    elif selectedLearner == 'LSVM':
        learnerParams = {'normalize': True}
    elif selectedLearner == 'Kmeans':
        learnerParams = {'n_clusters':8}
    elif selectedLearner == 'GMM':
        learnerParams = {'n_components':2, 'covariance_type':0}
    elif selectedLearner == 'SpectralCluster':
        learnerParams = {'n_clusters':8, 'eigen_solver':0}
    elif selectedLearner == 'ArgMax':
        learnerParams = {'sampling_rate':1}
    elif selectedLearner == 'ArgMin':
        learnerParams = {'sampling_rate':1}
    elif selectedLearner == 'DepthEstimate':
        learnerParams = {'sampling_rate':1}
    elif selectedLearner == 'LogR':
        learnerParams = {'penalty':0, 'C':1.0}
    elif selectedLearner == 'ANN':
        learnerParams = {'hidden_layer_sizes':(100,), 'activation':3, 'solver':2}
    return LearnerType(selectedLearner, learnerParams)




[docs]def modifyLearner(learner, newValue, pos=0):
    """Modifies a machine learner object

    Args:
        learner: learner to modify
        newValue: new value for learner parameter
        pos: which parameter to insert new value

    Returns:
        a learner object
    """
    learnerName = learner.learnerName
    if (learnerName == 'KNN') and (type(newValue) is int) and pos%2==0:
        learner.learnerParams['K'] = newValue
    elif (learnerName == 'KNN') and (type(newValue) is int) and pos%2==1:
        learner.learnerParams['weights'] = newValue
    elif (learnerName == 'SVM') and (type(newValue) is int):
        learner.learnerParams['kernel'] = newValue
    elif (learnerName == 'DNN') and (type(newValue) is int) and pos%2==0:
        learner.learnerParams['epochs'] = newValue
    elif (learnerName == 'DNN') and (type(newValue) is int) and pos%2==1:
        learner.learnerParams['batch_size'] = newValue
    elif (learnerName == 'RandForest') and (type(newValue) is int) and pos%3==0:
        learner.learnerParams['n_estimators'] = newValue
    elif (learnerName == 'RandForest') and (type(newValue) is int) and pos%3==1:
        learner.learnerParams['class_weight'] = newValue
    elif (learnerName == 'RandForest') and (type(newValue) is int) and pos%3==2:
        learner.learnerParams['criterion'] = newValue
    elif (learnerName == 'Boosting') and (type(newValue) is float) and pos%3==0:
        learner.learnerParams['learning_rate'] = newValue
    elif (learnerName == 'Boosting') and (type(newValue) is int) and pos%3==1:
        learner.learnerParams['n_estimators'] = newValue
    elif (learnerName == 'Boosting') and (type(newValue) is int) and pos%3==2:
        learner.learnerParams['max_depth'] = newValue
    elif (learnerName == 'Kmeans') and (type(newValue) is int):
        learner.learnerParams['n_clusters'] = newValue
    elif (learnerName == 'GMM') and (type(newValue) is int) and pos%2 ==0:
        learner.learnerParams['n_components'] = newValue
    elif (learnerName == 'GMM') and (type(newValue) is int) and pos%2 ==1:
        learner.learnerParams['covariance_type'] = newValue
    elif (learnerName == 'SpectralCluster') and (type(newValue) is int) and pos%2==0:
        learner.learnerParams['n_clusters'] = newValue
    elif (learnerName == 'SpectralCluster') and (type(newValue) is int) and pos%2==1:
        learner.learnerParams['eigen_solver'] = newValue
    elif (learnerName == 'ArgMax') and (type(newValue) is float):

    elif (learnerName == 'ArgMin') and (type(newValue) is float):
